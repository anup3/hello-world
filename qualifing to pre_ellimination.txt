#include<stdio.h>
int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
        long int n,k,i,j,t;
        scanf("%ld %ld",&n,&k);
        long int a[n];
        for(i=0;i<n;i++)
        {
            scanf("%ld",&a[i]);
            for(j=i;j>=1;j--)
            {
                if(a[j]>a[j-1])
                {
                   t=a[j];
                   a[j]=a[j-1];
                   a[j-1]=t;
                }
            }
        }
        while(a[k-1]==a[k])
        {
            k++;
        }
        printf("%ld\n",k);
    }
    return 0;
}
